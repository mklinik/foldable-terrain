\newcommand\vflap[4]
{
\draw (#1, #2)
  -- ++(#4 * \flapWidth, -\flapWidth)
  -- ++(0, -#3 + 2 * \flapWidth)
  -- ++(#4 * -\flapWidth, -\flapWidth)
  -- cycle
  ;
}
